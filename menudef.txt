OptionValue SevHUDSize
{
	11.0, "Always"
	12.0, "Automap Only"
}

OptionValue SevHUDStyle
{
	0, "0: Blank"
	1, "1: Standard"
	2, "2: Cells = Fuel"
	3, "3: FreeDoom"
	4, "4: Freestyle"
	5, "5: BTSX Episode 1"
	6, "6: BTSX Episode 2"
	7, "7: Neo Plutonia"
}

OptionValue SevHUDChexExtra
{
	0, "Vanilla"
	1, "Arms & Ammo"
}

OptionMenu "SevHUDOptionMenu"
{
	Title "$OPTMNU_SEVHUD"
	StaticText "Turn this off to use SevHUD"
	Option "$ALTHUDMNU_ENABLE",					"hud_althud", "OnOff"
	StaticText ""
	StaticText "Turn this off too -- Classic scaling is unsupported"
	Option "$SCALEMNU_HUD_OLDSCALE",			"hud_oldscale", "OnOff"
	StaticText ""
	StaticText "Make sure this doesn't say 'Unknown'"
	Option "Show SevHUD",				"screenblocks", 				"SevHUDSize"
	StaticText ""
	IfGame(doom)
	{
		StaticText "Choose the style that matches your loaded map pack"
		Option "$OPTMENU_SEVHUD_STYLE",			"sevhud_style", 			"SevHUDStyle"
		StaticText ""
	}
	IfGame(chex)
	{
		StaticText "See your arms and ammo, or stick to the vanilla style"
		Option "$OPTMENU_SEVHUD_CHEXEXTRA",			"sevhud_chexextra", 			"SevHUDChexExtra"
		StaticText ""
	}
	StaticText "Adjust to your preferred size"
	Slider "$SCALEMNU_HUD_SCALEFACTOR",			"hud_scalefactor", 0.0, 1.0, 0.025, 3, "hud_oldscale", 1
	StaticText ""
	StaticText "Open the automap and adjust until all erroneous lines disappear"
	StaticText "(If you often switch between multiple resolutions, ensure it looks correct in all of them)"
	Slider "HUD scale factor (Fine)",			"hud_scalefactor", 0.0, 1.0, 0.001, 3, "hud_oldscale", 1
	StaticText ""
	StaticText "If the line colors don't look good, fix them here"
	Submenu "$OPTMNU_AUTOMAP",			"AutomapOptions"
	StaticText ""
	StaticText "There's a custom crosshair, too, if you want it"
	Option "$HUDMNU_CROSSHAIR",				"crosshair", "Crosshairs"
	StaticText ""
	Option "$OPTMNU_SEVHUD_EXTRA", "sevhud_extra", "OnOff"
	Option "$OPTMNU_SEVHUD_HP_BAR", "sevhud_hpbar", "OnOff"
	StaticText ""
	StaticText "Check for updates at forum.zdoom.org/viewtopic.php?t=60201"
	StaticText ""
	StaticText "Check out Snap the Sentinel at snapgame.net!", "orange"
}
